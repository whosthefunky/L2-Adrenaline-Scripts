function  GetCity (shouldGoHome: boolean = true; inCityEvt: boolean = false): TCity;

  if User.InRange (116569, -179848, User.Z, 7000, 100)     then City := DWARVEN;
  if User.InRange (44904, 49800, User.Z, 7000, 100)        then City := ELVEN;
  if User.InRange (82876, 148960, User.Z, 5000, 100)       then City := GIRAN;

  if User.InRange (-14236, 123701, User.Z, 5000, 100)      then City := GLUDIO;
  if User.InRange (147713, -56202, User.Z, 5000, 100)      then City := GODDARD;
  if User.InRange (111434, 220426, User.Z, 5000, 100)      then City := HEINE;
  if User.InRange (117144, 76824, User.Z, 5000, 100)       then City := HUNTER;
  if User.InRange (-45080, -113576, User.Z, 5000, 100)     then City := ORC;
  if User.InRange (81150, 54880, User.Z, 5000, 100)        then City := OREN;
  if User.InRange (43838, -48222, User.Z, 5000, 100) or
    User.InRange (38283, -48102, User.Z, 5000, 100)        then City := RUNE;
  if User.InRange (88041, -142502, User. Z, 5000, 100)     then City := SHTUTTGART;
  if User.InRange (-83139, 243145, User.Z, 5000, 100)      then City := TALKING_ISLAND;

  // If it isn't in one of the previous locations then goHome and try again
  if (City = NON_CITY) and (shouldGoHome) then
  begin
    GoHome (-1, -1, inCityEvt);
    City := GetCity;
  end;
  Result := City;
end;

{
--------------------------------------------------------------------------------
  Different ways to return home
--------------------------------------------------------------------------------
}
procedure  GoHome (ItemID: integer = -1; skillID: integer = -1; move: boolean = false);
var
  Skill: TL2Skill;
  Item: TL2Item;
  Item2: TL2Item;
  Item3: TL2Item;
  City: TCity;
begin
  if  not User.Dead  then
  begin
    // Use the item past as parameter
    if  (itemID <> -1)  and  Inventory.User.ByID(itemID, Item)  then
    begin
      Engine.UseItem (itemID);
      Delay (User.Cast.EndTime + TPDelay);
    end;

    // Use skill to teleport if no item past as parameter
    if  (skillID <> -1)  and  (itemID = -1)  and  SkillList.ByID(skillID, Skill)  then
    begin
      Engine.UseSkill(skillID);
      Delay (User.Cast.EndTime + TPDelay);
    end;

    // Use SoE if no item or skill provided
    if  (skillID = -1)  and  (itemID = -1) and  Inventory.User.ByID(SCROLL_OF_SCAPE, Item)  then
    begin
      Engine.UseItem (Item);
      Delay (User.Cast.EndTime + TPDelay);
    end;

    City := GetCity (false, false);
    if  City = NON_CITY  then
    begin
      Print ('Couldn''t go home');
      exit;
    end;

    // Go to the Gatekeeper if move = true
    if  move  then MoveInCity;
  end;
end;

{
--------------------------------------------------------------------------------
  Returns the position in the array of the corresponding gatekeeper of the
  destination. Returns -1 if it's not found.
--------------------------------------------------------------------------------
}
function  getCityToLocationGK (teleportName: String): integer;
var
  i: integer;
begin
  Result := -1;

  while  Result = -1  do
  begin
    if  Pos (teleportName, GateKeepersStrings[i]) <> 0  then
    begin
      Result := i;
    end;
  end;
end;

{
--------------------------------------------------------------------------------
  It takes us to a city if we aren't in any. Then in an array of teleport lines
  we look for our destination, it will be equal to id of the city
  It returns true if the destination is the city in which we are
--------------------------------------------------------------------------------
}
function  isRightCity (destination: TCity; shouldGoHome: boolean = true): boolean;
begin
  Result := (GetCity(shouldGoHome) = destination);
end;

{
--------------------------------------------------------------------------------
  Give a description of this function
--------------------------------------------------------------------------------
}
function  toRightCity (destination: TCity): boolean;
var
  currentCity: TCity;
begin

  // If we are in our destination return true
  if  isRightCity (destination)  then
  begin
    Result := true;
  end else  // else we get the name of the city in which we are
  begin
    currentCity := GetCity;

    // If we aren't in a city we go home
    if  currentCity = NON_CITY  then  GoHome;

    // We take the ordinal value of the destination from the CityNames array and we teleport there
    TeleportTo (CityNames[ord(destination)]);

    // We repeat the process until we are in our destination
    Result := toRightCity (destination);
  end;
end;

{
--------------------------------------------------------------------------------
  Checks if the destination is actually the city to which we want to go.
--------------------------------------------------------------------------------
}
function  TeleportToLocation (destination: String): boolean;
begin
  Result := false;
  if  toRightCity(TCities[getCityToLocationGK(destination)])  then
    Result := TeleportTo (destination);
end;

{
--------------------------------------------------------------------------------
  Teleports to the destination
--------------------------------------------------------------------------------
}
function  TeleportTo (destination: String): boolean;
var
  City: TCity;
  gkId: integer;
  cityIndex: integer;
  gatekeeper: TL2Live;
  dialogIndex: integer;
  teleportIndex: TStringList;
  rightCity: boolean;
begin
  City := GetCity;
  cityIndex := Ord (City);
  Result := false;

  // If we are in a city and the length of the destination is not 0 then we continue
  if  (City <> NON_CITY)  or  (Length(destination) <> 0)  then
  begin
    teleportIndex := TStringList.Create;
    dialogIndex := 0;

    gkId := GateKeepers[cityIndex];  // Get the Gatekeeper ID of our current city

    {
      We specify the way in which the string is divided, in this case by commas.
      And we assign to teleportIndex the String with the locations of the gatekeeper
      of the current city
    }
    teleportIndex.Delimiter := ',';
    teleportIndex.DelimitedText := GateKeepersStrings[cityIndex];

    gatekeeper := nil;

    // search the gaketeeper by id
    NpcList.ById (gkId, gatekeeper);

    // if it was found we continue
    if  gatekeeper <> nil  then
    begin
      {
        We search the string destination in the possible destinations of our gatekeeper
        and assign it to dialogIndex. indexOf returns -1 if it wasn't found.
        We add 1 to match the dialog option of the gatekeeper
      }
      rightCity := true;
      dialogIndex := teleportIndex.indexOf (destination) + 1;
      if  dialogIndex = 0  then
      begin
        dialogIndex := 1;
          rightCity := false;
      end;
      {
        Checks if we are in the right city or teleport us to the first city of the dialog options
      }
      Engine.SetTarget (gatekeeper);
      Engine.MoveTo (gatekeeper);
      Engine.DlgOpen;
      Delay (500);
      // TODO: Check the gatekeepers that don't have the teleport dialog in the first option
      if  gatekeeper.ID = ROXXY  then  // if we are in talking island
        Engine.DlgSel (2)
      {
      else
        Engine.DlgSel (ifThen((Pos(buffer.name, 'Ragara') <> 0), 'Teleport', GkDlg));
        case buffer.id of
          30256: Engine.DlgSel(2);
          30006: Engine.DlgSel(3);
      }
      else
        Engine.DlgSel(1);
      Delay (500);
      Engine.DlgSel (dialogIndex);
      Delay (TPDelay);
      Result := true;

      if  rightCity = false  then
        TeleportTo (destination);
    end;
    teleportIndex.Free;
  end;
end;
